\section*{ПЕРЕЧЕНЬ УСЛОВНЫХ ОБОЗНАЧЕНИЙ}
\addcontentsline{toc}{section}{Список принятых сокращений}

\begin{description}
    \item[AI] — Искусственный интеллект
    \item[API] — Интерфейс прикладного программирования
    \item[GDELT] — Глобальная база данных событий, языка и тональности
    \item[GEOJSON] — Формат обмена пространственными данными на основе JSON
    \item[JTS] — Библиотека для манипулирования геоданными для языка программирования Java
    \item[LLM] — Большая языковая модель
    \item[ML] — Машинное обучение
    \item[NER] — Распознавание именованных сущностей
    \item[NLP] — Обработка естественного языка
    \item[OSM] — OpenStreetMap (Открытая геопространственная база данных)
    \item[REST] — Передача состояния представления
    \item[TRM] — Управление рисками в путешествиях
    \item[UML] — Унифицированный язык моделирования
    \item[VRAM] — Видеопамять
\end{description}
