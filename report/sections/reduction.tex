\section*{ПЕРЕЧЕНЬ УСЛОВНЫХ ОБОЗНАЧЕНИЙ}
\addcontentsline{toc}{section}{Список принятых сокращений}

\begin{description}
    \item[AI] — Artificial Intelligence (искусственный интеллект);
    \item[API] — Application Programming Interface (прикладной программный интерфейс);
    \item[GDELT] — Global Data on Events, Language, and Tone (глобальная база данных событий, языка и тональности);
    \item[GEOJSON] — Geographic JavaScript Object Notation (формат обмена пространственными данными на основе JSON);
    \item[JTS] — JTS Topology Suite (библиотека для манипулирования геоданными для языка программирования Java);
    \item[LLM] — Large Language Model (большая языковая модель);
    \item[ML] — Machine Learning (машинное обучение);
    \item[NER] — Named Entity Recognition (распознавание именованных сущностей);
    \item[NLP] — Natural Language Processing (обработка естественного языка);
    \item[OSM] — OpenStreetMap (открытая геопространственная база данных);
    \item[REST] — Representational State Transfer (передача состояния представления);
    \item[TRM] — Travel Risk Management (управление рисками в путешествиях);
    \item[UML] — Unified Modeling Language (унифицированный язык моделирования);
    \item[VRAM] — Video Random Access Memory (видеопамять).
\end{description}
